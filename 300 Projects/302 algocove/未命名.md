      public Response<Void> submitTeamCheckin(Long teamId, Integer
  problemId) {
          String userId = LoginUserContext.getUserId();
          LocalDate today = LocalDate.now();

          // 1. 检查是否是团队成员
          if (!teamMemberRepository.checkIsMember(teamId, userId)) {
              return Response.forbidden("不是团队成员");
          }

          // 2. 检查今日是否已完成团队打卡
          TeamCheckinRecord record = teamCheckinRecordRepository
              .selectByTeamUserDate(teamId, userId, today);

          if (record != null && record.getCompleted()) {
              return Response.badRequest("今日团队打卡已完成");
          }

          // 3. 记录打卡
          if (record == null) {
              record = new TeamCheckinRecord()
                  .setTeamId(teamId)
                  .setUserId(userId)
                  .setCheckinDate(today)
                  .setCompleted(true)
                  .setProblemIds("[" + problemId + "]");
              teamCheckinRecordRepository.insert(record);
          }

          // 4. 更新团队任务进度
          updateTeamTaskProgress(teamId, today);

          return Response.ok();
      }

      // 检查是否全员完成,发放奖励
      private void updateTeamTaskProgress(Long teamId, LocalDate date) {
          TeamCheckinTask task = getOrCreateTask(teamId, date);
          int completedCount = teamCheckinRecordRepository
              .countCompleted(teamId, date);

          task.setCompletedMembers(completedCount);

          // 全员完成
          if (completedCount >= task.getTotalMembers() &&
  !task.getAllCompleted()) {
              task.setAllCompleted(true);

              // 发放团队奖励
              List<TeamMember> members =
  teamMemberRepository.selectByTeamId(teamId);

                      task.getTeamReward()
                  );
