const startDate = dv.date("today").minus({ days: 7 });
const endDate = dv.date("today");

let entries = dv.pages('"Journal"')
    .where(p => p.file.day >= startDate && p.file.day <= endDate)
    .sort(p => p.file.day, 'asc');

for (let entry of entries) {
    let dayOfWeek = entry.file.day.toFormat("cccc");
    dv.header(3, `${dayOfWeek}, ${entry.file.day.toISODate()}`);
    dv.paragraph(entry.file.content);
}
